Intermediate

Work with classes and objects.


// Class and Object
class Car {
  String brand;
  Car(String b){ brand = b; }
  void drive(){ System.out.println("Driving " + brand); }
}

public class Main {
  public static void main(String[] args) {
    Car c = new Car("Tesla");
    c.drive();
  }
}
